Cookie 简单使用示范

2019-09-10 1 min read

先贴上代码，明天再写博客

@Controller
public class MainController {
    private static final String TAG="MainController";

    @RequestMapping("/")
    public String index(){
        return "index";
    }

    @ResponseBody
    @RequestMapping(value = "/cookies/{key}")
    public String getCookies(HttpServletRequest request, 
		@PathVariable String key){
        String result="未获取到sessionID";
        Cookie[] cookies = request.getCookies();
        if (cookies!=null){
            for (Cookie cookie:cookies){
                if ("sessionID".equals(cookie.getName())){
                    result = "sessionID:" + cookie.getValue();
                }
            }
        }
        return result;
    }
    @ResponseBody
    @RequestMapping(value = "/cookies", method = RequestMethod.POST)
    public String setCookies(HttpServletResponse response) {
        String result = "服务器出现错误";
        response.setStatus(500);
        try{
            Cookie cookie = new Cookie("sessionID","范鹏程");
            response.addCookie(cookie);
            response.setStatus(200);
            result = "cookie设置完成";
        }catch (Exception e){
            System.err.println(TAG+" :发生错误");
            System.err.println(e.getMessage());
        }
        return result;
    }
}


//向后台服务器发送请求cookie的请求
function setCookie() {
    $.ajax({
        url:"/cookies",
        type:"POST",
        success:function (result) {
            alert(result);
        }
    })
}

//向后台服务器发送字符串的同时获得cookie
function sendWithCookie() {
    $.ajax({
        url:"/cookies/sessionID",
        type:"GET",
        success:function (result) {
            alert(result);
        }
    })
}
